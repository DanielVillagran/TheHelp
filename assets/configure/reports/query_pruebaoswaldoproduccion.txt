SELECT [Movimientos].[FACTURADORA],[Movimientos].[NUMERO MOVIMIENTO],[Tipo Documento].[TIPO DOCUMENTO],[Clientes].[RAZON SOCIAL],[Clientes].[CURP],[Clientes].[RFC],[Documentos].[FOLIO],[Documentos].[FECHA],[Conceptos].[NOMBRE CONCEPTO],[Documentos].[TOTAL]*0.16 as IVA FROM facturacion.dbo.Movimientos  inner join facturacion.dbo.[Documentos] on [Movimientos].Facturadora=[Documentos].Facturadora AND [Movimientos].[ID DOCUMENTO]=[Documentos].[ID DOCUMENTO] inner join facturacion.dbo.[Tipo Documento] on [Documentos].Facturadora=[Tipo Documento].Facturadora AND  [Documentos].[ID TIPO DOCUMENTO]=[Tipo Documento].[ID TIPO DOCUMENTO] inner join facturacion.dbo.[Clientes] on [Documentos].Facturadora=[Clientes].Facturadora AND  [Documentos].[ID CLIENTE]=[Clientes].[ID CLIENTE] inner join facturacion.dbo.[Conceptos] on [Documentos].Facturadora=[Conceptos].Facturadora AND  [Documentos].[ID CONCEPTO]=[Conceptos].[ID CONCEPTO]  where Documentos.FACTURADORA LIKE '%VELTA%' AND Documentos.FECHA BETWEEN DATEADD(DAY, -5, getdate()) AND getdate()
